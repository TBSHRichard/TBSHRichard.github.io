----------------------------------------------------------------
-- Example showing the creation of an Event Property for a
-- MoonScript class. Look at Event.new for more info.
--
-- @classmod Foo
-- @author Richard Voelker
----------------------------------------------------------------

Event = require "Event"

class Foo
	new: =>
		@Changed = Event!

	-- Notify our observers!
	Change: =>
		@Changed\notifyObservers "bar"

	-- Our Event Property. It is set in the constructor.
	Changed: nil
